记忆化搜索
```python
N, M = map(int, input().split())  
memo = [0 for i in range(M+1)]  
# print(memo)  
for _ in range(N):  
    sign = 0  
    L = int(input())  
    temp = list(map(int, input().split()))  
    for i in range(L):  
        if memo[temp[i]] == 0:  
            memo[temp[i]] += 1  
            print(temp[i])  
            sign = 1  
            break  
    if sign == 0:  
        print('0')
```

# D

```python
def solve():  
    N = int(input())  
    A = list(map(int, input().split()))  
  
    # dp 字典存储以每个数字结尾的最长连续递增子序列长度  
    dp = {}  
    max_length = 0  
  
    for num in A:  
        # 如果存在 num-1，则可以接在后面  
        # 否则从当前数字开始新序列  
        dp[num] = dp.get(num - 1, 0) + 1  
        max_length = max(max_length, dp[num])  
  
    print(max_length)  
  
  
if __name__ == "__main__":  
    solve()
```
