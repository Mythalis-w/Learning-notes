
1. 遍历字典中的键
```python
dict_a = {'a':1 'b':2 'c':3 'd':4 'e':5}
# 遍历字典中的键
for key in dict_a:
	print(key)
	
# keys函数
for key in dict_a.keys():
	print(key)


# 获取字典中的值
for key in dict_a.keys():
	print(dict_a[key])

# values函数
for value in dict_a.values():
print(value)

# 想要同时获取key和value 需要用到items函数
for key, value in dict_a.items():
print(f'key:{key},value:{value}')
```

2. 遍历
```python
students = {
    'class1': [
        {'name': 'Alice', 'score': 85, 'age': 20},
        {'name': 'Bob', 'score': 78, 'age': 21}
    ],
    'class2': [
        {'name': 'Charlie', 'score': 92, 'age': 19}
    ]
}
for class_name, student_list in students.items():
    print(f"班级: {class_name}")
    for student in student_list:
        print(f"学生:{student['name']},分数:{student['score']},年龄: {student['age']}")
```
