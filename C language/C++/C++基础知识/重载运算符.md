
先看这一段代码
```C++
#include<iostream>
using namespace std;
class Per{
    public:
    int m_a;
    int m_b;
};
int main(){
    Per p1;
    p1.m_a=10;p1.m_b=20;
    
    Per p2;
    p2.m_a=10;p2.m_b=20;
    
    Per p3;
    p3.m_a=p1.m_a+p2.m_a;
    p3.m_b=p1.m_b+p2.m_b;
    
    cout<<p3.m_a<<endl;
    cout<<p3.m_b<<endl;
    return 0;
}
```
理解起来并不是什么难的
```C++
#include<iostream>
using namespace std;
class Per{
    public:
    int m_a;
    int m_b;
    Per paddr(Per &p){ //成员函数，以Per为返回类型
	    Per temp;
	    temp.m_a = this->m_a + p.m_a;
	    temp.m_b = this->m_b + p.m_b;
	    return temp;
	}
};
int main(){
    Per p1;
    p1.m_a=10;p1.m_b=20;
    
    Per p2;
    p2.m_a=10;p2.m_b=20;
    
    Per p3;
    p3=p1.paddr(p2)  // p1和p2去使用这个成员函数,以Per的形式返回，然后赋给p3
	
    cout<<p3.m_a<<endl;
    cout<<p3.m_b<<endl;
    return 0;
```
 然后我们对paddr这个方法名字改名为`operator+`
```C++
#include<iostream>
using namespace std;
class Per{
    public:
    int m_a;
    int m_b;
    Per operator+(Per &p){ //成员函数，以Per为返回类型
	    Per temp;
	    temp.m_a = this->m_a + p.m_a;
	    temp.m_b = this->m_b + p.m_b;
	    return temp;
	}
};
int main(){
    Per p1;
    p1.m_a=10;p1.m_b=20;
    
    Per p2;
    p2.m_a=10;p2.m_b=20;
    
    Per p3;
    p3=p1.operator+(p2)  // p1和p2去使用这个成员函数,以Per的形式返回，然后赋给p3
	
    cout<<p3.m_a<<endl;
    cout<<p3.m_b<<endl;
    return 0;
```

此时修改完后，我们就可以把 `p3=p1.operator+(p2)` 改为 `p3=p1+p2` ，这个就叫做运算符的重载
整个重载过程调用了3个类，在p1里去调用重载方法，并且p2传入，以temp暂时类返回赋值给p3
