```C
#include <stdio.h>

int main() {
    double eps;
    scanf("%le", &eps);
    
    double pi = 0.0;    // 存储π/2的近似值
    double term = 1.0;  // 当前项，初始为第0项: 1
    int i = 0;          // 项数索引
    
    // 累加直到最后一项小于eps
    while (term >= eps) {
        pi += term;           // 累加当前项
        i++;                  // 下一项
        term *= i * 1.0 / (2 * i + 1);  // 计算下一项: term * i/(2i+1)
    }
    
    // 如果最后一项小于eps，但还没有累加，需要累加一次
    pi += term;
    
    // 计算π的值: π = 2 * (π/2)
    pi *= 2;
    
    // 输出结果，保留5位小数
    printf("PI = %.5f\n", pi);
    
    return 0;
}
```
