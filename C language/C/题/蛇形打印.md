
```C
#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);
    
    int matrix[n][n];
    
    // 读取矩阵
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < n; j++) {
            scanf("%d", &matrix[i][j]);
        }
    }
    
    // 蛇形打印
    for(int i = 0; i < n; i++) {
        if(i % 2 == 0) {  // 偶数行：从左到右
            for(int j = 0; j < n; j++) {
                printf("%d ", matrix[i][j]);
                //if(i != n-1 || j != n-1) printf(" ");
            }
        } else {  // 奇数行：从右到左
            for(int j = n-1; j >= 0; j--) {
                printf("%d ", matrix[i][j]);
                //if(i != n-1 || j != 0) printf(" ");
            }
        }
    }
    
    return 0;
}
```
