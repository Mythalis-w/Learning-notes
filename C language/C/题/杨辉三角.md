
```C
#include<stdio.h>

int main(){
    int n;
    int arr[100][100];
    
    scanf("%d", &n);
    
    // 生成杨辉三角
    for(int i = 0; i < n; i++){
        // 每行的第一个和最后一个元素是1
        arr[i][0] = 1;
        arr[i][i] = 1;
        
        // 计算中间元素
        for(int j = 1; j < i; j++){
            arr[i][j] = arr[i-1][j-1] + arr[i-1][j];
        }
    }
    
    // 打印杨辉三角
    for(int i = 0; i < n; i++){
        for(int j = 0; j <= i; j++){
            printf("%d", arr[i][j]);
            if(j < i){
                printf(" ");  // 数字之间加空格，最后不空格
            }
        }
        printf("\n");
    }
    
    return 0;
}
```
