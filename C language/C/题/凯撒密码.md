```C
#include <stdio.h>
#include <string.h>

int main() {
    char str[81];  // 少于80个字符，加上\0需要81
    int offset;
    
    // 读取字符串（包含空格）
    fgets(str, sizeof(str), stdin);
    
    // 移除fgets可能读取的换行符
    int len = strlen(str);
    if(len > 0 && str[len-1] == '\n') {
        str[len-1] = '\0';
    }
    
    // 读取偏移量
    scanf("%d", &offset);
    
    // 处理负偏移量，转换为正偏移
    offset = offset % 26;
    if(offset < 0) {
        offset += 26;  // 将负偏移转换为正偏移
    }
    
    // 遍历字符串并加密
    for(int i = 0; str[i] != '\0'; i++) {
        char ch = str[i];
        
        if(ch >= 'a' && ch <= 'z') {
            // 小写字母加密
            printf("%c", 'a' + (ch - 'a' + offset) % 26);
        }
        else if(ch >= 'A' && ch <= 'Z') {
            // 大写字母加密
            printf("%c", 'A' + (ch - 'A' + offset) % 26);
        }
        else {
            // 非字母字符保持不变
            printf("%c", ch);
        }
    }
    
    printf("\n");
    return 0;
}
```
