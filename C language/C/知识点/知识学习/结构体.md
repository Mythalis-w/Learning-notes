
对于任意一个结构体，都需要声明 `struct` 才可以使用，且一般结构体的声明和函数一样，会放在主函数的前面，如

```C
#include <stdio.h>

struct Person {
    char name[50];  //定义成数组是为了储存字符串
    int age;
    float height;
};

int main() {
    // 使用结构体（必须带 struct 关键字）
    struct Person p1;  // 定义一个变量，使用这个结构体,相当于分别为p1定义了三个变量
    struct Person p2 = {"张三", 25, 175.5}; //这个相当于直接定义的p2，并已经为p2定义
    
    // Person p3;  // ❌ 错误！不能省略 struct
    
    p1.age = 30; //使用的时候用定义的 变量名. 去使用就好了
    strcpy(p1.name, "李四");
    
    return 0;
}
```

对于变量的定义还可以在定义结构体的时候就直接定义好

```C
struct Person {
    char name[50];
    int age;
    float height;
} p1, p2;
```

这样就直接定义好了

## 结构体数组

```C
#include <stdio.h>

struct Student {
    int id;
    char name[20];
    float score;
};

int main() {
    // 初始化结构体数组
    struct Student students[3] = {   //定义数组
        {1001, "张三", 85.5},
        {1002, "李四", 92.0},
        {1003, "王五", 78.5}
    };
    
    struct Student *p;  // Student结构体的指针
    p = students;
    
    // 求平均成绩
    float sum = 0;
    for(int i = 0; i < 3; i++) {
        sum += p[i].score;  // 或者 sum += (p+i)->score
    }
    float average = sum / 3;
    
    printf("学生成绩列表：\n");
    for(int i = 0; i < 3; i++) {
        printf("%d %s %.1f\n", 
               students[i].id, 
               students[i].name, 
               students[i].score);
    }
    
    printf("\n平均成绩：%.2f\n", average);
    
    return 0;
}

```

通过这种方式，可以实现方便的求平均

还有不同的写法


1. 使用数组下标方式
```C
p = students;
float sum = 0;
for(int i = 0; i < 3; i++) {
    sum += p[i].score;  // 像数组一样使用指针
}
```

2. 使用指针算术
```C
p = students;
float sum = 0;
for(int i = 0; i < 3; i++) {
    sum += p[i].score;  // 像数组一样使用指针
}
```

3. 直接移动指针
```C
p = students;
float sum = 0;
for(int i = 0; i < 3; i++) {
    sum += p->score;
    p++;  // 指针移动到下一个元素
}
// 注意：此时p已经指向数组末尾了
```

4. 使用指针遍历
```C
struct Student *ptr;
float sum = 0;
for(ptr = students; ptr < students + 3; ptr++) {
    sum += ptr->score;
}
```

5. 指针偏移
```C
p = students;  // p指向数组开头
float sum = 0;
for(int i = 0; i < 3; i++) {
    sum += (p + i)->score;  // 使用偏移，不改变p本身
}
// 循环结束后：p 仍然指向 students[0]
// 还可以继续使用 p 访问数组
```
