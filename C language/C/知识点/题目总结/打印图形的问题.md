
最简单的情况，即字符和空格都连在一起，那只需定义三个变量，i、j、h，一个代表空格，一个表字符，一个代表行数，将i、j与行数联系起来就可以了
(比如，第一行三个空格，一个星号，第二行2个空格，三个星号，则空格和行数之和恒为4，星号与行数为2倍减1的关系)
# 三角形

如下:打印上等腰三角形

```C
#include<stdio.h>
int main(){
    for(int i=1;i<=3;i++){          // 外层循环，控制行数，共3行
        for(int q=0;q<3-i;q++){     // 内层循环1：打印前导空格
            printf(" ");            // 空格数量 = 3 - 当前行数
        }
        for(int c=0;c<2*i-1;c++){   // 内层循环2：打印星号
            printf("*");            // 星号数量 = 2*当前行数 - 1
        }
        printf("\n");               // 换行，进入下一行
    }
    return 0;
}
```

结果为

```
  *
 ***
*****
```

在这其中，我们定义i是行数，第一个for，即3行，第二个for 是去打印空格的，第三个for是去打印* 的，如此就能做到打印一个等腰三角形，如果想打印n行的，将3换为n，再用 scanf 函数读取就可以，如下。

```C
#include<stdio.h>
int main(){
	int n;                          // 定义变量n，用于存储用户输入的三角形高度
	scanf("%d",&n);                 // 从用户输入读取三角形的高度
	
    for(int i=1;i<=n;i++){          // 外层循环：控制行数，从第1行到第n行
        for(int q=0;q<n-i;q++){     // 内层循环1：打印前导空格
            printf(" ");            // 空格数量 = n - 当前行数i
        }                           // 使三角形在控制台中居中对齐
        
        for(int c=0;c<2*i-1;c++){   // 内层循环2：打印星号
            printf("*");            // 星号数量 = 2*当前行数i - 1
        }                           // 形成等腰三角形的两边
        
        printf("\n");               // 换行，准备打印下一行
    }
    return 0;
}
```

结果为（n=5）

```
    *
   ***
  *****
 *******
*********
```

而倒三角形则仅需反过来即可，空格第一行0个 第二行1个，字符第一行2n-1，第二行2n-3

```C
#include<stdio.h>
int main(){
    int n;                          // 定义变量n，用于存储用户输入的数字
    scanf("%d",&n);                 // 从用户输入读取n的值
    
    for(int i=1;i<=n;i++){          // 外层循环：控制行数，从第1行到第n行
        for(int k=1;k<=i-1;k++){    // 内层循环1：打印前导空格
            printf(" ");            // 空格数量 = 当前行数i - 1
        }                           // 每行前面的空格数逐行增加
        
        for(int q=1;q<=2*(n-i+1)-1;q++){  // 内层循环2：打印星号
            printf("*");            // 星号数量 = 2*(n-i+1)-1 = 2n-2i+1
        }                           // 星号数量逐行递减
        
        printf("\n");               // 换行，准备打印下一行
    }
    return 0;
}
```

结果为（n=5）

```
*********
 *******
  *****
   ***
    *
```
# 菱形

对于菱形，其实就是在三角形的基础上再加上一个倒三角

```C
#include<stdio.h>
int main(){
    // 上半部分：正三角形（3行）
    for(int i=1;i<=3;i++){          // 外层循环：控制上半部分行数，共3行
        for(int q=0;q<3-i;q++){     // 内层循环1：打印前导空格
            printf(" ");            // 空格数量 = 3 - 当前行数i
        }                           // 使三角形居中对齐
        for(int c=0;c<2*i-1;c++){   // 内层循环2：打印星号
            printf("*");            // 星号数量 = 2*当前行数i - 1
        }                           // 星号数量逐行递增：1,3,5
        printf("\n");               // 换行
    }
    
    // 下半部分：倒三角形（2行）
    for(int i=1;i<=2;i++){          // 外层循环：控制下半部分行数，共2行
        for(int q=1;q<=i;q++){      // 内层循环1：打印前导空格
            printf(" ");            // 空格数量 = 当前行数i
        }                           // 空格数量逐行递增：1,2
        for(int k=1;k<=5-i*2;k++){  // 内层循环2：打印星号
            printf("*");            // 星号数量 = 5 - 2*当前行数i
        }                           // 星号数量逐行递减：3,1
        printf("\n");               // 换行
    }
    return 0;
}```

结果为
```
  *
 ***
*****
 ***
  *
```

如果想要打印n行的菱形，则只需要定义一个行数（一般情况n为奇数）

```C
#include<stdio.h>
int main(){
    int n;                          // 定义变量n，用于存储用户输入的数字
    scanf("%d",&n);                 // 从用户输入读取n的值
    int hang=(n+1)/2;               // 计算菱形的上半部分行数（包括中间行）
                                    // 使用(n+1)/2确保无论是奇数还是偶数都能正确处理
    
    // 上半部分：正三角形（包括中间行）
    for(int i=1;i<=hang;i++){       // 外层循环：控制上半部分行数
        for(int q=0;q<hang-i;q++){  // 内层循环1：打印前导空格
            printf(" ");            // 空格数量 = hang - 当前行数i
        }                           // 使菱形居中对齐
        for(int c=0;c<2*i-1;c++){   // 内层循环2：打印星号
            printf("*");            // 星号数量 = 2*当前行数i - 1
        }                           // 星号数量逐行递增：1,3,5,7,...
        printf("\n");               // 换行
    }
    
    // 下半部分：倒三角形（不包括中间行）
    for(int i=1;i<=hang-1;i++){     // 外层循环：控制下半部分行数
        for(int q=1;q<=i;q++){      // 内层循环1：打印前导空格
            printf(" ");            // 空格数量 = 当前行数i
        }                           // 空格数量逐行递增
        for(int k=1;k<=2*hang-i*2;k++){  // 内层循环2：打印星号
            printf("*");            // 星号数量 = 2*hang - 2*当前行数i
        }                           // 星号数量逐行递减
        printf("\n");               // 换行
    }
    return 0;
}
```

# X形

对于X形图案，可以根据输出的行数，打印一个n\*n的的空格，然后在代码里用if来判断是否输入字符

```C
#include<stdio.h>
int main(){
    int n;
    scanf("%d",&n);
    
    for(int i=1;i<=n;i++){          // 控制行数
        for(int k=1;k<=n;k++){      // 控制列数
            if(k==i || k==n+1-i)    // 判断是否在对角线上
                printf("X");
            else 
                printf(" ");
        }
        printf("\n");               // 换行
    }
    return 0;
}
```

结果为

```
(n=5)
X   X
 X X 
  X  
 X X 
X   X

(n=6)
X    X
 X  X 
  XX  
  XX  
 X  X 
X    X
```

# 平行四边形

平行四边形则更为简单

```C
#include<stdio.h>
int main(){
    int n, m;                       // 定义变量：n-行数，m-每行星号数量
    scanf("%d %d", &n, &m);         // 从用户输入读取行数和每行星号数
    
    for(int i = 1; i <= n; i++){    // 外层循环：控制行数，从第1行到第n行
        for(int k = 1; k <= i - 1; k++)  // 内层循环1：打印前导空格
            printf(" ");            // 空格数量 = 当前行数i - 1
        
        for(int c = 1; c <= m; c++) // 内层循环2：打印星号
            printf("*");            // 每行固定打印m个星号
        
        printf("\n");               // 换行，进入下一行
    }
    return 0;
}
```

结果是（输入 4 5）

```
*****
 *****
  *****
   *****
```
