def main() -> None:

    train_df = pd.read_csv("train.csv")

    test_df = pd.read_csv("test.csv")

    train_df = train_df.dropna(subset=["num_sold"]).copy()
	'''- `dropna(subset=["num_sold"])`：删除 `num_sold` 为空的行（目标值缺失的样本不能用于监督训练）
- `.copy()`：生成一个独立副本，避免后续链式赋值引发 `SettingWithCopyWarning` 或潜在副作用'''
  
    x_train, y, x_test, train_dates = build_features(train_df, test_df)

    y_log = np.log1p(y).astype(np.float32)# 对数化处理，避免负数和0的影响
	# np.log1p(y) = log(1+y)
  

    seeds = [42, 2024, 3407]

    pred_logs = []

    val_scores = []

    for seed in seeds:

        pred_log, best_rmsle = train_one_seed(seed, x_train, y_log, train_dates, x_test)

        pred_logs.append(pred_log)

        val_scores.append(best_rmsle)

  

    pred_log = np.mean(np.stack(pred_logs, axis=0), axis=0)
    '''1. `np.stack(pred_logs, axis=0)`  
    把每个 seed 的预测向量堆成一个矩阵。  
    假设有 3 个 seed、测试集 98550 条，那么形状是：

- 每个 seed：`(98550,)`
- stack 后：`(3, 98550)`

第 0 维就是“seed 维度”。

2. `np.mean(..., axis=0)`  
    沿 seed 维度求均值，得到每条测试样本的平均预测：'''
    

    pred = np.expm1(pred_log)
    # expm1(x)=e^x−1
	# 把log空间还原回去
    pred = np.maximum(pred, 0)
    # 把负值改为0，销量没有负
	
  

    submission = pd.DataFrame({"id": test_df["id"], "num_sold": pred})

    submission["num_sold"] = submission["num_sold"].round(3).astype(np.float32)

    submission.to_csv("submission_basic_mlp.csv", index=False)

  

    print(f"Seeds: {seeds}")

    print(f"Best Val RMSLE per seed: {[round(s, 6) for s in val_scores]}")

    print(f"Mean Val RMSLE: {np.mean(val_scores):.6f}")

    print("预测完成，已保存: submission_basic_mlp.csv")

    print(submission.head(10).to_string(index=False))
