def build_features(

    train_df: pd.DataFrame, test_df: pd.DataFrame

) -> tuple[np.ndarray, np.ndarray, np.ndarray, pd.Series]:

  

    y = train_df["num_sold"].to_numpy(dtype=np.float32) #取出我们要预测的值num_sold

    train_dates = pd.to_datetime(train_df["date"], errors="coerce") # 把train_df["date"]这一列转为to_datetime然后赋值给train_dates

  

    x_train_df = add_date_features(train_df.drop(columns=["num_sold"]).copy())
    # 这个是我们要传入的train，删去num_sold(y去储存，在后面参与损失的计算)，然后经过日期特征展开后的训练特征 DataFrame,复制一份赋值给 x_train_df

    x_test_df = add_date_features(test_df.copy())
	# 同上
  

    full = pd.concat([x_train_df, x_test_df], axis=0, ignore_index=True)
    # 把 `x_train_df` 和 `x_test_df` 按行上下拼接成一个总表 `full`, `axis=0`：按行拼（竖着拼), `ignore_index=True`：拼完后重新编号索引（0,1,2,...）

    full = pd.get_dummies(

        full,

        columns=["country", "store", "product"],

        dummy_na=False,

    ).fillna(0.0)
    
    # 第一个参数传入表，第二个参数是选取列，第三个参数是 在 one-hot 时，不为缺失值 NaN 单独创建一列。 .fillna(0.0)是吧NaN转为0.0
	# pd.get_dummies()是一个one—hot的函数，one—hot举例子就是：比如说有一列三个商店，会变成三列分别代表三个商店，然后用1代表是哪一个，0代表不是，将字符串转化为 1 0
  

    x_all = full.to_numpy(dtype=np.float32)
    # 同上转为numpy

    x_train = x_all[: len(x_train_df)]
    x_test = x_all[len(x_train_df) :]
	# 这两行就是把full -> x_all 的train和test数据切片分开
  

    mean = x_train.mean(axis=0, keepdims=True)
    std = x_train.std(axis=0, keepdims=True)
    std[std < 1e-6] = 1.0
    x_train = (x_train - mean) / std
    x_test = (x_test - mean) / std
	# 最后这五行是处理输入数据的，让输入数据更加的集中，然后对处理过的x_test数据根据y去输出（不需要还原X，我们只是改变了训练数据的标准，输出并没有变）
  

    return x_train, y, x_test, train_dates
