
```python
def solve():
    N = int(input())
    A = list(map(int, input().split()))
    
    # dp 字典存储以每个数字结尾的最长连续递增子序列长度
    dp = {}
    max_length = 0
    
    for num in A:
        # 如果存在 num-1，则可以接在后面
        # 否则从当前数字开始新序列
        dp[num] = dp.get(num - 1, 0) + 1
        max_length = max(max_length, dp[num])
    
    print(max_length)

if __name__ == "__main__":
    solve()
```
