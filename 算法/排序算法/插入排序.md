```C
#include <stdio.h>

void printArray(int arr[], int n) {
    for (int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
}

int main() {
    int n;
    scanf("%d", &n);
    
    int arr[n];
    
    // 读取数组
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    
    // 输出原始序列
    printf("原始序列：");
    printArray(arr, n);
    printf("\n");
    
    // 插入排序，并输出每一趟结果
    for (int i = 1; i < n; i++) {
        int key = arr[i];  // 当前要插入的元素
        int j = i - 1;
        
        // 找到插入位置并移动元素
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = key;  // 插入元素
        
        // 输出当前趟排序结果
        printf("第%d趟排序结果：", i);
        printArray(arr, n);
        printf("\n");
    }
    
    return 0;
}
```

```C
void a(int arr[], int n){  //n是总元素个数
	for(int i=1;i<n;i++){
		int key=arr[i]; //key是将要排序的元素
		int j=i-1; //j是开始比较的元素
		while(j>=0 && arr[j]>key){ //相当于判断是否还需要向左插入，循环结束代表找到值
			arr[j+1]=arr[j];  //不断向上覆盖，参考上面的图
			j--;
		}
		arr[j+1]=key;
	}
}
```
