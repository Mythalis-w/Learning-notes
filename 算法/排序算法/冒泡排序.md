# C
 对于C语言，没有像py中一样有sort函数，无法通过函数排序,所以要通过一类算法来解决这个问题

```C
#include<stdio.h>
int main(){
    int arr[1000000];
    int n,temp;
    scanf("%d",&n);
    for(int i=1;i<=n;i++){
        scanf("%d",&arr[i-1]);
    }
    for(int i=0;i<n-1;i++){
        for(int k=0;k<n-1;k++){
            if(arr[k]<arr[k+1]){
                temp=arr[k];
                arr[k]=arr[k+1];
                arr[k+1]=temp;
            }
        }
    }
    return 0;
}
```

这是一个降序排序，即让最大值放在最前面。
对于降序排序，为了优化算法，我们可以从后往前比较（如果从前往后进行比较，最后一个就是最小值，找最大值就很麻烦，需要遍历结束，因此按情况讨论就可以），保证第一轮arr\[0]就是最大值，这样可以缩小第三个for的遍历次数，如下

```C
#include<stdio.h>
int main(){
    int arr[1000000];
    int n,temp;
    scanf("%d",&n);
    for(int i=1;i<=n;i++){
        scanf("%d",&arr[i-1]);
    }
    for(int i=0;i<n-1;i++){
        for(int k=n-2;k<=0+i;k--){  //主要是这里做了改动
            if(arr[k]<arr[k+1]){
                temp=arr[k];
                arr[k]=arr[k+1];
                arr[k+1]=temp;
            }
        }
    }
    return 0;
}
```

对于升序排序，即最大值在最后面，对第一个程序进行改动,就能得到升序排序的算法

```C
#include<stdio.h>
int main(){
    int arr[1000000];
    int n,temp;
    scanf("%d",&n);
    for(int i=1;i<=n;i++){
        scanf("%d",&arr[i-1]);
    }
    for(int i=0;i<n-1;i++){
        for(int k=0;k<n-1-i;k++){
            if(arr[k] > arr[k+1]){
                temp=arr[k];
                arr[k]=arr[k+1];
                arr[k+1]=temp;
            }
        }
    }
    return 0;
}
```

不论升序还是降序，我们都能再进行优化，添加标识符

```C
#include<stdio.h>
int main(){
    int arr[1000000];
    int n,temp,sign;
    scanf("%d",&n);
    for(int i=1;i<=n;i++){
        scanf("%d",&arr[i-1]);
    }
    for(int i=0;i<n-1;i++){
	    sign=0;
        for(int k=0;k<n-1-i;k++){
            if(arr[k]>arr[k+1]){
                temp=arr[k];
                arr[k]=arr[k+1];
                arr[k+1]=temp;
                sign=1;
            }
        }
        if(sign==0) break;
    }
    return 0;
}
```

通过标识符，可以在排序完成后提前终止遍历。

写成函数如下

```C
void bubbleSort(int arr[], int n) {
    int temp, sign;
    for(int i = 0; i < n - 1; i++) {
        sign = 0;
        for(int k = 0; k < n - 1 - i; k++) {
            if(arr[k] > arr[k + 1]) {
                temp = arr[k];
                arr[k] = arr[k + 1];
                arr[k + 1] = temp;
                sign = 1;
            }
        }
        if(sign == 0) break;
    }
}
```

以后对于任何数组，执行这个函数就可以完成排序。
