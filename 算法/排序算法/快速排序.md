
```C
#include<stdlib.h>
//这个函数头是必须的

// 用户定义的比较函数
int compare_int(const void *a, const void *b) {
    return *(int*)a - *(int*)b;  //整数就是升序，负数就是降序，也可以用if中return1，-1
}

int main() {
    int arr[] = {5, 2, 8, 1};
    
    // 第一个数组名，第二个数组个数，第三个占的字节数
    qsort(arr, 4, sizeof(int), compare_int);
    //                            ↑
    //    这里传递的是函数地址，不是调用函数,在qsort中的某个地方调用
    
    return 0;
}
```
