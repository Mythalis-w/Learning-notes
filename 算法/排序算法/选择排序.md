```C
#include <stdio.h>
// 选择排序函数
void selectionSort(int arr[], int n) {  //等价于void selectionSort(int *arr, int n)
    int i, j, minIndex, temp;
    
    // 遍历所有数组元素
    for (i = 0; i < n - 1; i++) {
        // 假设当前未排序部分的第一个元素索引为最小值
        minIndex = i;
        
        // 在剩余的未排序部分中寻找最小值
        for (j = i + 1; j < n; j++) {
            if (arr[j] < arr[minIndex]) {
                minIndex = j;
            }
        }
        
        // 将找到的最小值与当前位置的元素交换
        if (minIndex != i) {
            temp = arr[i];
            arr[i] = arr[minIndex];
            arr[minIndex] = temp;
        }
    }
}

int main() {
    int arr[1000];
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    selectionSort(arr,n);
    for(int i = 0; i < n; i++) {
    printf("%d ", arr[i]);
	}
    return 0;
}
```

这就是一个选择排序，通过不断地遍历整个数组来找出最小的数值的索引，最后进行交换。
