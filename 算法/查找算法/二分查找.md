#include<stdio.h>
int main(){
    int n,arr[10000],result;
    scanf("%d %d",&n,&result);
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    
    int signLeft=0,signRight=n-1;
    int mid=(signLeft+signRight)/2; //mid=signLeft+(signRight-signLeft)/2 可以防止大数相加造成整数溢出，即注释里的是更好的条件
    
    while(signLeft<=signRight){
        if(arr[mid]>result){
            signRight=mid-1;  //mid已经比较过了
            mid=(signLeft+signRight)/2;
        }
        else if(arr[mid]<result){
            signLeft=mid+1; //mid已经比较过了
            mid=(signLeft+signRight)/2;
        }
        else{
            printf("%d",mid);
            break;
        }
    }
    return 0;
}
